# liunx 系统的命令

# <<开放端口>>

        
# 这个意思就是开放 8080 端口，以便外部访问 8080 端口的服务
firewall-cmd --zone=public --add-port=8080/tcp --permanent
# 这个意思就是关闭 8080 端口，外部就访问不了 8080 端口的服务了
firewall-cmd --zone=public --remove-port=8080/tcp --permanent
# 重新加载配置，这样开放的端口才能生效
firewall-cmd --reload
# 查看所有开放的端口
firewall-cmd --zone=public --list-ports

# <<防火墙命>>

        
# 电脑重启
reboot   
# 查看防火墙的状态，打印的信息如有  active(running) 就表示启动中
systemctl status firewalld.service
# 临时关闭防火墙，一旦电脑重启后，防火墙就要启动了。打印的信息如有  inactive(dead) 就表示关闭了
systemctl stop firewalld
# 临时启动防火墙，如果防火墙的状态是默认关闭的，，一旦电脑重启后，防火墙就是关闭的，打印的信息如有  active(running)  就表示启动中
systemctl start firewalld
# 永久关闭防火墙，电脑重启后，防火墙还是关闭状态
systemctl disable firewalld
# 永久启动防火墙，电脑重启后，防火墙还是启动状态
systemctl enable firewalld

# <<服务状态>>

# 在 centsOs 7 默认是没有 netstat 命令, 需要安装 yum install -y net-tools 
        
# 查看服务是否启动，如果有服务占用了 5001 端口，就会显示出来 
netstat -lnpt|grep 5001

# 在 centsOs7 环境安装 ipfs

# <<IPFS 命令集>>

        
# 查看 ipfs 版本，如果显示就证明安装成功了
ipfs version  
# 查看 ipfs 节点信息，包含了节点哈希
ipfs id   
# 显示配置文件中的内容
ipfs config show   
# 执行这俩步命令，可以进入 ipfs 的默认仓库，这里存放配置文件、数据、api 链接等信息
cd ~ cd .ipfs/  
# 启动 ipfs 节点
ipfs daemon

#   安装 IPFS

        
# 1 安装 wget 下载工具
yum -y install wget  
# 2 在根目录创建 ipfs 目录，以便存放下载资源
mkdir ipfs   
# 3 切换创建的目录
cd ipfs   
# 4 下载 IPFS 提供的 liunx 版本 tar 包
wget https://github.com/ipfs/go-ipfs/releases/download/v0.4.23/go-ipfs_v0.4.23_linux-amd64.tar.gz/  
# 5 解压 tar 包
tar zxvf go-ipfs_v0.4.23_linux-amd64.tar.gz   
# 6 解压后会有 go-ipfs 文件夹，切换到这个文件夹
cd go-ipfs    
# 7 安装 ipfs
./install.sh    
# 8 验证安装是否成功，如果成功输出版本号，就表示安装好了
ipfs version 
# 9 初始化节点
ipfs init   
# 1 这个是输出 ipfs 节点信息
ipfs cat /ipfs/{CID}/readme

# 文件上传到 ipfs

        
# 使用这个命令，可以把文件夹，或者一个文件添加到 ipfs
ipfs add -r ./ipfs/ 
        
    
# 输出

[root@localhost ipfs]# ipfs add -r  ./ipfs/
 added QmUNLLsPACCz1vLxQVkXqqLX5R1X345qqfHbsf67hvA3Nn ipfs/name.txt
 added QmX3GXDfVEG5HRChoiAstYKjiZuAkwKojaPhzExvMiY4U3 ipfs
 0 B / ? [------------------------------------------------------------------------

        
# 这是把文件夹或者文件得 CID 发送到 ipfs 上，这一步成功了就可以通过网关访问到这个资源了
ipfs name publish QmX3GXDfVEG5HRChoiAstYKjiZuAkwKojaPhzExvMiY4U3 
        
    
# 输出

[root@localhost ipfs]# ipfs name publish QmX3GXDfVEG5HRChoiAstYKjiZuAkwKojaPhzExvMiY4U3
Published to QmcUuccBvYjTN1mj8oteqZMDqj6q7SRFFhytgbi683HpcJ: /ipfs/QmX3GXDfVEG5HRChoiAstYKjiZuAkwKojaPhzExvMiY4U3

        
# 查看 ipfs 节点固定得所以 CID, 可以通过这样的方式，验证上次到 ipfs 的文件夹或者文件是否成功
ipfs name resolve [节点得 CID ]
