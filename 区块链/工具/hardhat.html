# 创建项目

npm init
npm install --save-dev hardhat
npx hardhat
npm install --save-dev @nomiclabs/hardhat-waffle ethereum-waffle chai @nomiclabs/hardhat-ethers ethers

# 多文件开源:

  多文件的合约开源，需要借助 hardhat 工具来完成，在浏览器中很能实现多个文件开源

# 1. 安装依赖
  "@nomiclabs/hardhat-etherscan": "^3.1.0",                    // 合约开源工具
  "@openzeppelin/contracts": "^4.7.3",                         // 代码库            
  "undici": "^5.10.0"                                          // 代理配置
# 2. 配置文件 hardhat.config.js
// 开源的工具包，合约开源需要借助这个工具包
require("@nomiclabs/hardhat-etherscan");

// 为 hardhat 工具配置代理，不然总是连接超时，这个需要注意
const { ProxyAgent, setGlobalDispatcher } = require("undici");
const proxyAgent = new ProxyAgent('http://127.0.0.1:7891');   // 这里就是本地的代理端口
setGlobalDispatcher(proxyAgent);


/** @type import('hardhat/config').HardhatUserConfig */
module.exports = {

  // 配置编译器版本，以及编译参数  
  solidity: {
    version: '0.8.4',
    settings: {
      optimizer: {
        enabled: true,
        runs: 200
      }
    }
  },

  // 配置网络
  networks: {
    bscTestnet: {
      url: 'https://nd-518-217-842.p2pify.com/f7d1bc5feaebcdf4f03186435c7949be',
      accounts: ['3601d568b8436e97cc5e9a80b7e2e43d7095784b3002cd23a4662aa3a65c70ca']
    }
  },

  // 配置合约开源参数
  etherscan: {

    // 开源多个文件的合约需要使用到 BINANCE 浏览器生成的一个 KEY, 
    apiKey: {
      // 每一个网络对于的一个 KEY  
      bscTestnet: 'F53ANW6FZUXDT333YCQ6HJR36P8HDT62QK'
    },

    // 这里配置的是合约要开源对于的链的信息。
    customChains: [
      {
        network: "bscTestnet",
        chainId: 97,
        urls: {
          apiURL: "https://api-testnet.bscscan.com/api",
          browserURL: "https://testnet.bscscan.com"
        }
      }
    ]

  }
};
# 3. 开源命令以及参数
[contracts/DispatcherV2.sol:TestContracts]   : 入口合同路径, 这个参数添好即可. 程序会自动引入依赖的文件
[--network bscTestnet]                       : 要把合约开源到那个网络
[0x06f25145fea1594947f72eC1782581614a60e170] : 开源的合约地址

[                                          ] : 如果开源的合约有构造参数，按照构造参数的顺序在后面追加即可

npx hardhat verify --contract contracts/DispatcherV2.sol:TestContracts --network bscTestnet 0x06f25145fea1594947f72eC1782581614a60e170
