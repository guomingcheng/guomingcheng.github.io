# 标准

# 以太链上发行的代币，也称为 erc-20 代币，包含了下面这些 public 接口，这个合约才是是标准的 rec-20 代币

pragma solidity 0.7.5;

interface IERC20{
    /**
     *  类型：合约接口 
     *  数量：9
     *  说明：当一个合约有了一下这些方法，那么这个合约就是标准的 ERC-20 合约
     * 
     */
    function name() public view returns(string memory);  //全称
    function symbol() public view returns(string memory); //简称
    function decimals() public view returns(uint); //代币的可分割的位数
    function totalSupply() public view returns(uint) //总量
    function balaceOf(address acount) public view returns(uint); //返回发送者持有多少代币
    function allowance(address owner, address spender) public view returns(uint);  //查询 owner 授权给 spender 多少代币
    function transfer(address recipient, uint amount)public returns(bool);//转账功能，发送者向 recipient 地址转入 amount 数量的代币，成功返回 true
    function approve(address spender, uint amount)public returns(bool); //授权功能，发送者向 spender 授权 amount 数量的代币，一旦授权，spender 就能转走发送者的代币
    //中介者函数，sender 授权了给 发送者，发送者可以把 sender 的地址转到 recipient 地址
    function transferFrom(address sender, address recipient, uint amount) public returns(bool);
    
}

# ERC20 实现

pragma solidity 0.7.5;


import "./Content.sol";   //用于获取上下文的 msg.sender
import "./IERC20.sol";         //ERC 20 标准的接口
import "./SafeMath.sol";   //安全数学库


contract ERC20 is Content, IERC20{

    string private _name;
    string private _symbol;
    uint256 private _decimals = 18;
    uint256 private _totalSupply;
    
    using SafeMath for uint256; // 导库, SafeMath 库内定义的方法，都可以被 uint256 类型字段调用
    
    mapping(address => uint256) private _balances; //映射每个地址的代币数量
    mapping(address => mapper(address => uint256)) private _allowances; //授权映射，第一个地址向第二个地址授权 N 个代币，一旦授权了，第二个地址就可以转移第一个地址的代币

    event Transfer(address indexed from, adddress indexed to, uint256 value);
    event Approval(address indexed owner, address indexed spender, uint256 value);
    
    constructor(string memory name, string memory symbol, uint256 totalSupply)public{
        _name = name;
        _symbol = symbol;
        _totalSupply = totalSupply;
    }

    //return : name
    function name() public view override returns(string memory){
        rteurn _name;
    }   
    
    //return: symbol
    function symbol() public view override returns(string memory){
        return _symbol;
    }
    
    //return: decimals
    function decimals() public view override returns(uint){
        return _decimals;
    }
        
    //return: totalSupply
    function totalSupply() public view override returns(uint){
        return _totalSupply;
    }
    function balancesOf(address account) public view override returns(unit256){
        return _balaces[account]; // 返回 account 地址持有代币的总量
    }

    function allowance(address owner, address spender) public view override returns(uint256){
        return _allowances[owner][spender];
    }

    function transfer(address to, uint256 amount) public  override returns(bool){
        address owner = _msgSender();
        _transfer(owner, to , amount);
        return true;
    }
    
    function approve(address spender, uint256 amount) public override return(bool){
        address owner = _msgSender();
        _approve(owner , spender , amount);
        return true;
    }
