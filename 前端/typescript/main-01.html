# 基元类型

# 基元类型是 Ts 提供了静态编程语言中的类型识别，和 java 基本类型一样，也是拥有着 string 字符串、number 整数、number[] 数组、 {} 对象类型

// 这意思是声明一个 string 类型的 name 变量, 在 ts 中对这种声明称为《类型注释》
let name:string = "guomingcheng"
let value:number = 100                  // 这里同理
let arr: number[] = [1, 2,3]            // 这是声明一个 number 类型的数组
let obj = {                             // 这是声明一个对象的类型，也是 json 对象，这里不需要类型注释，    
    x: 1,                               // 对象内部的变量可以不声明类型注释，ts 会根据赋给的值自动的推断出来
    y: 1
}

// 我们在函数中的参数也要明确一个参数的类型，这里就明确了 proo 是一个 string, 调用 greet 函数不能随便的传参了，必
// 须传入的是 string 类型
function greet(proo: string ): string{  // 函数的尾部 : 冒号的类型，表示的是函数的返回类型，如果不明确返回的
                                        // 类型， ts 就会自动的推断出函数返回的类型

    // `` 反引号是 ES6 的语法，称为模板字符串，在 `` 包围内可以使用 ${} 方式引入其他变量，`` 自动的字符串拼接
    // 我认为他的优势就是可以随意的位置字符串拼接
    console.log(`proo ${proo}`)
}
// 函数的调用
greet("adsa")

# 联合类型

# 联合类型是一个类型或者多个类型的组合被称为联合类型， 多个类型之间使用 | 符号分隔，表示一个参数可以接受 number 类的字面量或者接受 string 字面量
但是：要使用联合类型时，要坐进一步的处理，因为 ts 是一个严格的编译检查，他是无法知道你是 number 还是   string 类型
   // 这就是联合类型的使用方式，vue 参数是一个 string 类型，也可以是一个 number 类型
   let vue: string|number = "xixi"


   // last?：string , 这里加了个 ? 问号，表示这个参数是可选的，可以选择传入或者选择不传入
   // 在 TS 这个写法就有点相当于 java 的方法重载，相同函数名称与不同的参数个数，但是在函数内使用的 last 需要坐进一步的处理
   function pintName(obj:{fist: string, last?: string}){

       // 注意，这种可传参的参数，不能 obj.last 使用，因为在 ts 检查时，last 不一定时 string 类型的，也可能是 unf 的
       // 所有使用需要 obj.last? ,加一个 ? 问号，这表示 last 有参数的话，就输出，否则就不输出
       //console.log(obj.last.charAt(1));

       // 另一种的使用方式，if 判断
       if(obj.last != undefined){      // undefined 是一个类型，不是字符串，这个需要注意
       }
   }
   //这里参数传入的对象就只有一个类型，就可以

# 类型别名

# 定义类型的别名，就是给基元类型、联合类型、对象等，去一个别名，这个别名与所赋值的类型是一个等价的，一般用于有些长度的类型取一个短的别名
// 这个就为联合类型定义了一个别名，可以使用 Point， 因为 Point 具备了 number|string 联合类型的状态
type Point = number|string

// 这个就是为对象类型定义一个别名，与上面相同，也是可以定义其他的类型的别名
type Obj = {
    x: number,
    y: number
}

# 接口类型

# 接口类型与 type 定义的别名功能差不多，只不过接口类型是可以通过 extends 继承其他的接口类型，获取父接口内的定义的变量
// ts 的接口定义好像不用函数，只是仅仅的定义字段 ，感觉和声明一个对象差不多
interface Points{

    x: number,
    y: number
}
// 因为接口本来也是一种类型，所以在这里就使用接口 Points 类型作为参数
function pringCoords(pt: Points){

}
pringCoords({x:100,y:20})  // 使用

// 这里的 Bear 类型就继承了 Points 内的 x: number 与 y:number 类型
interface Bear extends Points{
    z: number
}
