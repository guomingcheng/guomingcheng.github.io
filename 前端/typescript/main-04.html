# 对象类型

在 Ts 中，对象类型分别是《匿名对象》使用 {} 定义的类型，这个是没有对象名称的，《接口对象》使用 interfact 关键字来定义，这个是有对象名称的， 《别名对象》是使用 type 关键字来为一个匿名对象取一个别名，这个也算是有对象名称的。都是通过这三种方式表达一个对象类型。

// 这是匿名对象的类型定义
function fu(obj: {num:number}){

}
// 这是接口定义的对象类型
interfact obj{
    num:number
} 

// 这是为匿名对象取一个名称
type obj = {
    num:number
}

// 这三种定义方式表达了对象类型，同样，这是三种发送也是等价的

# 属性操作权限


//定义对象
interface SomeType{
    // ? 这里的问号与在函数形参定义的一样，表示这个参数是一个可选的
    value?: string, 
    // 使用 readonly 关键字标记属性是只读的，无法修改，对于基元类型值是无法修改，对于对象类型，无法修改指向对象的指针，但是可以修改指向对象内部的属性的值
    readonly prop: string,
    // 这里也是使用了可读属性
    readonly home: {prop: string}
}
function doSomeThing(obj: SomeType){

    //obj.prop = "无法修改属性的值，因为在对象种定义了这个属性只能是一个可读的"

    //obj.home = {prop: "name"}     // readonly  可读标识，表示 obj.home 无法重新指向一个新的对象
    obj.home.prop = "可修改的"       // 但是可以修改对象内部属性的值   

}

# 索引签名

这是 TS 提供了一种对象中的属性定义方式，仅当无法确定定义的对象类型要包含那些属性时，就可以使用这种索引签名。个人认为这与可变参数有些类型，我 称为可变属性，实例化的时候可以添加一个或者多个属性都可以

interface DOg{
    // 这个就是定义可变属性的例子，[key:string] 表示左边 key 是字符类型，key类型只能是 string或number 类
    // number 表示的是可变属性的类型，也约束这个对象定义的属性类型必须都是 number
    [key: string]: number
    value: number           // 这样是可以的，因为 value 是 number,
    //name:string           // 这样是不行，上面的可变属性约束的类型并不兼容 string 类型
}

// 使用了这个对象，实列的 key-value 可以是一个或者多个
let test: DOg = {name: 18, usm: 17, }


// 这个 key 如果设置为 number 类型的，实列就是以一个数组的方式赋值
// 但是要注意，虽然他是这样赋值，但是他还是一个对象，他无法是数组的函数，这个是需要注意的一点
interface DOg{
    [key: number]: number,
}

let test: DOg = [1,3,4]

# 属性扩展

在 TS 中也提供了俩种方式实现属性扩展，一种是通过继承的方式来实现属性的扩展，一种是通过定义别名的方式扩展

interface Colorful{
    color: string
}
interface Circle{
    reduis: number
}

// 接口扩展，要通过 关键字继承父对象的属性，这样 ColorCircle 对象就有了上面俩个属性了
interface ColorCircle extends Colorful, Circle{

}

// 这个就是通过别名的方式实现扩展，通用继承了上面俩个属性
type ColorCircle = Colorful & Circle
