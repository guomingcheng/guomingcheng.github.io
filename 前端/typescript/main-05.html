# 泛类类型

学到此处，我发现在 TS 种所以的都行都可以称为类型，连这次的泛型函数，也是一种类型

// 一个泛型的函数实现，identity<T> 这个 T 是用来推断用户的类型，调用时不写，就根据传入参数的类型来推断 T 的类型
function identity<T>(arg: T): T{
    return arg;
}

// myIdentity 变量的类型就是一个泛型函数类型，上面实现的函数可以赋值给 myIdentity
let myIdentity1: <T>(arg: T) => T = identity
// 这里我就不明白了，这个泛型函数类型不是对象的一个属性吗，为何 identity 也可以赋值呢，真时搞不懂
let myIdentity2: { <T>(arg: T): T } = identity

interface Generic<T>{
    (agr: T): T
}
// 我是真的不明不白，唉
let myIdentity3: Generic<string> = identity

# 泛型类

Ts 中的类，与 java 中的类并没有多大的差异

class GenericNumber<T>{
    value: T
    add: (x: T, y: T) => T
}

// 创建泛型类的对象实例
let myGener: GenericNumber<number> = new GenericNumber<number>()
// 修改类属性的值
myGener.value = 1;
myGener.add = (x , y) => x+y

# 泛型约束

interface Lengthwise{
    length: number
}
// 他这个泛型约束与java 的泛型边界概念一样
// 这样约束告示调用者，传入的类型必须具备了 Lengthwise 对象中的属性，现在 Lengthwise 中的属性是 length
// 所以，调用者必须传入的类型，必须具备了 length 属性
function loggingIdentity<T extends Lengthwise>(agr: T): number{
    return agr.length
}


// 这个需要拆分才好理解
// <key extends T> 这个我们知道是约束参数的传入，key 参数必须是具备了属性，或者类型符合才可以
// <keyof T> 这意思是， T 是一个对象类型， 而 keyof 是只能接受对象类型，获取 T 对象中的所有属性 Key, 并转为联合的文字类型，
function getProperty<T , Key extends keyof T>(obj: T, key: Key){

}

// 这里有个对象的实例，比如 keyof x, 获取 X 所有的 KEY 值并转换为文字联合类型，就是 "a"|"b"|"c"
let x = {
    a: 1,
    b: 2,
    c: 3
}
// 第二个参数值，被约束了只能是 x 对象中的 Key 
getProperty(x , "a")
getProperty(x , "b")
