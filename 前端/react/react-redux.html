# 概念

# redux 的出现也是为了更好的状态管理，他的作用类似于 react 的上下文 contonex 传递，我也不清楚 uniswap 为何选择使用 redux, 为了看懂源码，我也过来学习

# redux 三个核心概念
#   1、store , 存储应用程序的所以的状态信息，全部交给 redux 管理
#   2、action , 描述状态发生变化的动作
#   3、reducer, 根据 action 描述的动作，执行状态变化
# redux 编写三个原则
#   1、redux 希望整个应用程序的状态都存储在单个 {} 对象树中，就是一个对象嘛，再交给 redux store 管理
#   2、更改状态的唯一方法是发出一个动作，这个一个对象，有一个 type 属性来描述更改那个

# redux 实现

# 当要使用 redux 使用状态的管理时，我们先在看看在 js 程序中是是如何表现的，必须要清楚 redux 所具备了的概念

# 安装依赖包 redux


import { legacy_createStore } from "redux"
// combineReducers() 函数是可以把多个 reducer 组合在一起，在进行构建
 
// redux 动作描述
const BUY_CAKE = 'BUY_CAKE'

// 当前函数返回一个对象，这个对象必须包含一个 type 属性，来描述修改状态的动作
function buyCake(){
  return {
    type:  BUY_CAKE,
    info: 'Firt redux action'
  }
}

// 初始化状态的之，必须是一个对象的类型，也将是提交给 redux 来管理  
const initalState = {
  numOfCakes: 10
}

// reducer 函数相当 react hooks 的 useStaet() , 具备的功能是返回状态，以及修改状态的函数
// 但是 reducer 区别在于，state 修改的动作需要在 reducer 函数中定义，比如，如果只是定义了状态值只能 +，那莫这个 state 就不能修改
// reducer 返回的是一个状态
const reducer = (state = initalState, action) => {
    // 根据动作的描述来修改状态
    switch(action.type){
      // 这里的状态更新也是与 react 一样，返回一个新的状态对象
      case BUY_CAKE : return { numOfCakes: state.numOfCakes - 1}

      // 次初始化的时候，才作为新值
      // 就算后面调用 dispatch() 函数，执行到这一步，也不会作为新值
      default:
        console.log("进来了") 
        return state;
    }
}

// redux 管理构建状态的 legacy_createStore 函数，接受词一个参数，这个参数就是 reducer() , 他包含了初始化状态对象，以及能修改状态的动作
const store = legacy_createStore(reducer);
// 获取 redux 管理的状态对象
console.log("inital state ", store.getState());

// 订阅函数，一但状态出现修改，就会回调这个参数的函数，
// 这个订阅函数有一个返回值，返回这个值也是函数，可以用于取消当前订阅的消息
const unsubscribe = store.subscribe(() => {console.log('updated state ', store.getState())})

// 可以设置多条订阅
// store.subscribe(() => {console.log('updated state 1239 ', store.getState())})

// 修改状态，根据动作来描述状态的修改
store.dispatch(buyCake());
store.dispatch(buyCake());

// 取消订阅
unsubscribe()
store.dispatch(buyCake());
store.dispatch(buyCake());
console.log("inital state ", store.getState());

# reducer 组合构建 state

# 以目前来看，这个方式应该不是那么常用
//  redux 中如何使用俩个或者多个 reducer 组合构建 state 状态
// 这个需要都如 combineReducers() 函数
import { legacy_createStore, combineReducers } from "redux"

// redux 动作描述
const BUY_CAKE = 'BUY_CAKE'
const BUY_ICECREAM = 'BUY_ICECREAM'

// 描述动作
function buyCake () { return { type: BUY_CAKE } }
function buyIceCream () { return { type: BUY_ICECREAM } }

const initalCakeState = { numOfCakes: 10 }
const initalIceCreamState = { numOfIceCreams: 20 }

const cakeReducer = (state = initalCakeState, action) => {
    switch (action.type) {

        case BUY_CAKE: return {
            numOfCakes: state.numOfCakes - 1
        }
        default:
            console.log("进来到 cakeReducer")
            return state
    }
}

const iceCreamReducer = (state = initalIceCreamState, action) => {
    switch (action.type) {

        case BUY_ICECREAM: return {
            numOfIceCreams: state.numOfIceCreams + 1
        }
        default:
            console.log("进来到 iceCreamReducer")
            return state
    }
}

// 使用 combineReduceers() 函数组合 reducer，
// 这样就可以组合多个 reducer，key 值是可以自定义的，

// 初始化的时候调用 3 次 reducer 函数，但是 dispatch() 函数的时候，只调用对应的 reducer, 目前来还行
const rootReducer = combineReducers({
    cake: cakeReducer,
    iceCream: iceCreamReducer
})

// 这一步，也是和上一次的使用一样了，这不过根据 rootReducer 作为参数来创建
const store = legacy_createStore(rootReducer)
store.subscribe(() => { console.log('updated state ', store.getState()) })

store.dispatch(buyCake())
store.dispatch(buyCake())
store.dispatch(buyIceCream())
store.dispatch(buyIceCream())

# redux-logger

# 这个是 redux 的日志功能，他这个日志可以非常直观的看到一个状态的变量，应该是一个很好用的，需要安装 redux-logger 包才行
// applyMiddleware 这个函数是中间件应用
// 需要安装 redux-logger 包，才能使用状态改变打印
import { legacy_createStore, applyMiddleware } from "redux"
import {logger} from 'redux-logger'

const BUY_CAKE = 'BUY_CAKE'

function buyCake(){
  return {
    type:  BUY_CAKE,
  }
}
const initalState = { numOfCakes: 10}


const reducer = (state = initalState, action) => {
    // 根据动作的描述来修改状态
    switch(action.type){
      // 这里的状态更新也是与 react 一样，返回一个新的状态对象
      case BUY_CAKE : return { numOfCakes: state.numOfCakes - 1}
      default:
        console.log("进来了") 
        return state;
    }
}

// 设置 applyMiddleware() 中间件函数, 这个函数的参数是 logger 
// 设置了 applyMiddleware() 中间件函数，一旦状态发生修改，参数的 logger 都会把时间、久状态、动作、新状态都会打印出来
const store = legacy_createStore(reducer, applyMiddleware(logger));
// 获取 redux 管理的状态对象
console.log("inital state ", store.getState());

store.dispatch(buyCake());
store.dispatch(buyCake());

# redux 的异步操作

# 这一步感觉异常的复杂，这个程序是，在执行修改状状态执行异步操作，根据返回的数据在进行状态修改。在这里需要安装 redux-thunk 与 axios 包

import { legacy_createStore, applyMiddleware } from "redux"
import thunk from "redux-thunk"
import axios from "axios"

const initalState = {
    loading: false,
    users: [],
    error: ''
}

// 状态的动作描述
const FETCH_USERS_REQUEST = 'FETCH_USERS_REQUEST'
const FETCH_USERS_SUCCESS = 'FETCH_USERS_SUCCESS'
const FETCH_ISERS_FAILURE = 'FETCH_ISERS_FAILURE'

// 请求描述
const fetchUsersRequest = () => {
    return {
        type: FETCH_USERS_REQUEST
    }
}

// 成功描述, 这里有个知识点，原来这个函数不仅描述动作，还可以接受参数
// axios
// thunk 是 react 的生态一个包，这是一个中间件，用于定义异步标准
// redux-thunk 安装
const fetchUsersSuccess = (users) => {
    return {
        type: FETCH_USERS_SUCCESS,
        payload: users
    }
}

const fetchUsersFailure = (error) => {
    return {
        type: FETCH_ISERS_FAILURE,
        payload: error
    }
}

const reducer = (state = initalState, action) => {

    switch (action.type) {

        case FETCH_USERS_REQUEST:
            return {
                ...state,
                loading: true
            }
        case FETCH_USERS_SUCCESS:
            return {
                loading: false,
                users: action.payload,
                error: ''
            }
        case FETCH_ISERS_FAILURE:
            return {
                loading: false,
                users: [],
                error: action.payload
            }

        default: return state
    }
}

const fetchUsers = () => {

    // 传递，dispatch 参数等价与 store.dispatch() ，可以通过这个 dispatch 参数修改状态
    return function (dispatch) {

        // 这里根据动作的描述修改状态，等价于 store.dispactch(fetchUsersRequest()) 这样修改
        dispatch(fetchUsersRequest())

        // 异步调用
        axios.get("https://jsonplaceholder.typicode.com/users")
            // url 请求成功，回调这个函数，返回的数据
            .then(response => {
                console.log(response)
                const users = response.data.map(item => item.id)
                dispatch(fetchUsersSuccess(users))
            })
            // 请求出现错误，回调这个函数，并传递错误信息
            .catch(error => {
                dispatch(fetchUsersFailure(error))
            })
    }
}

// 通过 applyMiddlware(thunk) 中间件的对接，可以让 dispactch() 方法接收一个函数，而不是动作，可以在这个函数内执行一些副作用
// 如异步任务，等等的副作用调用。
// 在这个函数内也是可以修改 redux 管理的状态，通过传递进来的 dispatch 参数
const store = legacy_createStore(reducer, applyMiddleware(thunk))

store.subscribe(() => { console.log(store.getState()) })

// dispatch 函数，也是接受一个函数，这个函数必须是返回一个函数，可以在这个函数内通过传递的 dispatch()函数来修改 redux 保存的状态
// dispatch 函数，需要使用 redux-thunk 包，配合才能使用
store.dispatch(fetchUsers())

# react-redux

# 如果在 react 集成 redux 的状态管理，需要下载 react-redux 包
# 这一步要使用到 Provider 标签，store 属性的值就是我们构建好的 redux store 状态，这里的 store 是由 ./react-redux/reducer' 文件提供
import React from 'react'
import ReactDOM from 'react-dom'
import { Provider } from 'react-redux'
import App from './App'
import AppCombine from './App_combineReducers'
import { store } from './react-redux/reducer'


// 渲样根组件 APP, 到一个 id 为 root 的 dom 节点上
ReactDOM.render(
  <React.StrictMode>
    <Provider store={store}>
      <App />
    </Provider>
  </React.StrictMode>,
  document.getElementById('root')
)
# 这一步就是子组件中读取状态，或者修改状态的值
import { useDispatch, useSelector } from "react-redux"
import { buyCake } from "./react-redux/cake/cakeActions"


function App () {
    
  // 多个 reducers 构建的状态是如何取值, 通过构建时的 key 值访问，注意，这个也只能 key 指向对象的一个属性
  // dispacth() 函数还是和以前一样使用
  // useSelector(state => state.key.count)

  // useSelector 函数调用一次是返回状态一个属性的值，他这个与 store.getState() 函数是有区别的，这个是需要注意
  const count = useSelector(state => state.count)

  // 这一步相当于 store.dispatch() 函数，把动作描述对象传给 dispatch() 就可以修改 redux 管理的
  const dispatch = useDispatch()

  return (
    <div >
      <p>{count}</p>
      // 根据动作来修改 redux 管理的对象，这个一步和以前是一样的
      <button onClick={() => { dispatch(buyCake()) }}>点击</button>
    </div>
  )
}

export default App

# redux-devtools

# 这是一个浏览器的插件，安装了这个插件，就可以追踪 store 状态的变化，但是在代码中通过 redux-devtools-extension 来支持，插件才能追钟当前应用程序的状态变化
import { combineReducers, legacy_createStore, applyMiddleware } from "redux"
import { composeWithDevTools } from "redux-devtools-extension"
import logger from "redux-logger"
import { caeamReducer } from "./caeam/caeamReducer"
import { cakeReducer } from "./cake/cakeReducer"


const reducers = combineReducers({
  cake: cakeReducer,
  caeam: caeamReducer,
})

// 想要插件追踪应用程序的状态变化，需要这句代码支持 composeWithDevTools(applyMiddleware(logger))
export const store = legacy_createStore(
    reducers, 
    composeWithDevTools(applyMiddleware(logger))
  )
