# 函数组件

# 函数组件定义的约定

# 1、组件的名称必须是大写，react 内部会根据这个来判断是一个函数组件还是普通的 html 标签
# 2、函数组件必须有返回值，表示该组件的 UI 结构，如果不需要渲染任何的内容，则返回 null
# 3、渲染时，使用函数的名称，进行 <Nmae/> 或者 <Name><Name/>
// 函数组件

// 创建函数组件
function Hello(){
  return <div>创建函数组件</div>
}

function App() {
  return (
    <div className="App">
      { /* 渲染函数组件, 名称也必须是与定义的函数名一样，还有一种可以是这样 <Hello><Hello/> */}
      <Hello/> 
    </div>
  );
}

export default App;

# 类组件


// 类组件
import React from "react";

// 需要继承 React.Component， 重写 render 函数，返回的值就是 UI 结构
class HelloComponent extends React.Component{
  
  render(){
    return <div>类组件</div>
  }
}

function App() {
  return (
    <div className="App">
      // 渲染方式是与函数一模一样，只不过 class 是通过类名
      <HelloComponent/>
    </div>
  );
}

export default App;

# 绑定事件

# 语法写作方式：

# 1、on + 事件名称={事件处理函数} ， 比如： <div onClick={()=>{}}>
# 2、react 事件采用的是驼峰命民法，移入事件: onMouseEnter； 聚焦事件: onFocus

// 类组件
import React from "react";

// 需要继承 React.Component， 重写 render 函数，返回的值就是 UI 结构
class HelloComponent extends React.Component{

  // 事件处理函数, 这样也是标准的写法，避免 this 指向问题
  clickHandler = (e) => {  // 这个 e 对象，事件触发后传进来，这个 e 包含了这个事件的所以内容
    // e.preventDefault()  // 这个意思是阻止事件标签的默认行为
    console.log("事件点击")
  }
  
  render(){
    // 监听 div 标签的点击事件
    // 注意: 如果事件触发，需要向处理函数传入参数，需要改造成 ()=> this.clickHasndler("处理"); ,这里就能向处理函数传值了
    return <div onClick={this.clickHandler}>类组件</div>
  }
}

function App() {
  return (
    <div className="App">
      // 渲染方式是与函数一模一样，只不过 class 是通过类名
      <HelloComponent/>
    </div>
  );
}

export default App;

# 组件状态

# 一个组件能够通过状态来影响行为，这个在 react hook 出来之前，函数式组件是没有自己的状态的，所以我们都是通过类组件来演示
# 步骤: 初始化状态 ==》 读取状态 ==》 修改状态 ==》 影响视图
# 注解： react 的状态管理与 vue 的数据绑定都是实现了同样的效果，渲染属性的值如果动态发生变化，那么渲染层次绑定该属性的位置也发送变化，在 react 中，函数组件没有属性值，所以状态管理组件只能通过类组件来实现

// 类组件
import React from "react";

// 需要继承 React.Component， 重写 render 函数，返回的值就是 UI 结构
class HelloComponent extends React.Component{

  // 状态管理的值是一个对象，但是这个变量命必须是 state
  state = {
    name: "guomingcheng",
  }

  // 事件处理函数
  clickHandler = () => {
    // 注意：在 react 中是不能直接通过赋值来修改，必须通过一个 setStart() 方法修改属性的值，才能实现动态渲染
    // 而 setStart() 方法是继承了 React.Component 模板类，得来的
    this.setState({
      // 动态修改
      name: "修改了======"
    })
  }
  
  // this 指向问题
  // render 函数中的 this 是被 react 内部做了修正，指向的就是当前的组件的实例对象
  // 箭头函数是没有 this 的，他会直接沿用上级的 this, 而这个上级就是 render 函数中的 this, 所以箭头函数内的 this 也是指向当前组件的实例对象      
  render(){
    // 监听 div 标签的点击事件
    return (
      <div>
        {this.states.name}
        // this.clickHandler 是一个箭头函数
        <button onClick={this.clickHandler}>点击</button>
      </div>
    )
  }
}

function App() {
  return (
    <div className="App">
      // 渲染方式是与函数一模一样，只不过 class 是通过类名
      <HelloComponent/>
    </div>
  );
}

export default App;

# 组件 state 改变

# 当一个组件的状态某一个类型的值，要修改时，是不能通过直接赋值的方式来修改状态。而是居于继承 React.Component 类提供的 setState() 方法来修改状态的值，这样才可以。所以，我们要知道实际开发中是如何修改某系类型的值
// 在这个状态下，定义类常见类型的状态，来表示如果修改
state = {
    count: 0,
    list: [1, 2, 3],
    person: {name: 'jsak', age: 18}
}

// 值类型修改
clickHandler = ()=>{
    
    // 这样就可以修改值类型的状态
    this.setState({ 
        count : this.state.count++
    })

    // 修改数组的状态，数组增加一个值
    this.setState({ 
        // 这就是网上说的 react 的问题，数组的状态修改需要重新的变量整数数组，就如下面所写
        // 他先拿到状态中的久值，并使用张开运算符，在后面添加一个值，就形成了往状态数组中添加一个值
        list: [...this.state.list, 4]
    
        // 删除
        // 这个 filter 函数表示筛选，如果回调函数返回 flase, 那么这个下标元素就不会在返回的数组内
        list: this.state.list.filter((item) => item != 2)    

    })

    // 修改对象的状态，对象更新一个属性
    this.setSTate({
        // 他这个也是先拿到状态的久值，并使用张开运算符，在后面添加一个属性，这个属性的值要更新的值，所以原来的 name 的值更新为 roes
        person: { ...this.state.person, name: 'roes'}
    })
}

# input 受控组件


// 类组件
import React from "react";

// 构建 input 受控组件
class HelloComponent extends React.Component{

  // 消息状态
  state = {
    message: "guomingcheng",
  }

  // 双向绑定处理函数
  clickHandler = (e) => {
    
    console.log(e.target.value)
    this.setState({
      // 通过 e.target.value 可以，获取 input 文本框的内容
      message: e.target.value
    })
  }
  
  render(){
    // input 组件
    return (
      // value 绑定状态的 message，就能实现 vue 的双向绑定
      <input type="text" value={this.state.message}
        // 为这个 input 组件，绑定一个变动事件，文本框一但变动就会触发
        onChange={this.clickHandler}
      />
    )
  }
}

function App() {
  return (
    <div className="App">
      <HelloComponent/>
    </div>
  );
}

export default App;
