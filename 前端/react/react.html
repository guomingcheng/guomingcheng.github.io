# jsx 介绍

# jsx 是 javasrcipt HTML 的缩写，表示在 js 代码中写 html 文本结构，这也是 react 的文件后缀，在 react 中创建 html 结构（UI 界面），并充分的利用 js 自身的能力编程动态的数据绑定
# 注意: jsx 并不是标准的 js 语法，他是 js 语法的扩展，浏览器默认是不支持识别的，所以我们要安装 @babel/plugin-transform-recat-jsx 包才能解析。这个包 create-react-app 脚手架已经内置了

# jsx 表达式

{} 表达式用于渲染变量值，和简单的表达式；


可用于表达式的类型有: 
string number bool null undefined object [] {} (function 类型) 函数调用

这写都是可以渲染 出来的

let name = "guomingcheng"

function App() {
  return (
    <div className="App">
      { name }
    </div>
  );
}

export default App;

# jsx 列表渲染


//技术方案：使用 {} 表达式，在使用 songs map 函数遍历并返回 li 标签，就可以实现列表渲染

const songs = [
  {id: 1, name: "qwe"},
  {id: 2, name: "asd"}
]

function App() {
  return (
    <div className="App">
      //
      <ul>
        // 通过 {} 表达式实现列表渲染
        // li 标签也是和 vue 使用一样，需要一个唯一的 id 值，不过 key 不会出现在真实的 dom 结构中，只在内部使用
        {songs.map((song) => <li key={song.id}>{song.name}</li>)}
      </ul>
  
    </div>
  );
}

export default App;

# jsx 条件渲染


//条件渲染
//技术方案：三元表达式、逻辑 && 运算


const flag = false

function App() {
  return (
    <div className="App">
     // 这个条件渲染，也是通过 {} , 如果 flag 为 true, 就会渲染 span 标签，否则就什么都不渲染
     // 注意：这个也是可以渲染复杂的标签，比如：模板，都是可以的
     { flag ? <span>guomingcheng</span>: null}
  
     // && 逻辑表达，如果为 true, 就渲染
     { true && <span>true</span>}
    </div>
  );
}

export default App;

# jsx 样式处理

//样式渲样
//内联样式 - 在元素标签上绑定一个 style 属性即可，和原生的类似
//类名样式

import "./app.css"  // 导入样式

const flag = false

function App() {
  return (
    <div className="App">
      // 这里注意：使用了俩个 {} , 第一个 {} 是为了把内部的 {} 渲染成对象，因为 style 属性是要包含多个值
     <span style={{color:'red'}}>das</span>

      // 这里使用了 className 属性，用于绑定样式 class
     <span className="active">测试类名绑定</span>
    
      // 动态绑定   
     <span className={ false ? "active" : null}>动态类名样式绑定</span>   
    </div>
  );
}

export default App;

# jsx 的注意事项

# 1、jax 必须有一个根标签，如果没有根标签，可以使用 <></> (幽灵标签) 把多组标签包裹起来，这样也算是一个根标签
# 2、所以的标签必须形成闭合，闭合就是 "<span></span>" 就是这样，或者 <img />
# 3、jsx 是支持多行 html 标签，如果需要换行编写，建议使用 （） 包裹起来，提高可读性
